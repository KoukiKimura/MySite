## 2. エントリーポイント

### 2.1 main.js

アプリケーション全体の初期化を行うエントリーポイント。

```javascript
// web/src/main.js

import { Router } from './shared/router/router.js';
import { Header } from './shared/components/header.js';
import { Footer } from './shared/components/footer.js';
import { PageContainer } from './shared/components/page-container.js';
import './shared/styles/main.css';

/**
 * アプリケーション初期化
 *
 * 実行順序:
 * 1. DOM構造を構築（header → page-container → footer）
 * 2. Routerを初期化
 * 3. 初回ルート解決
 */
export function initApp() {
  const app = document.getElementById('app');

  const header = Header.create();
  const pageContainer = PageContainer.create();
  const footer = Footer.create();

  app.appendChild(header);
  app.appendChild(pageContainer);
  app.appendChild(footer);

  Router.init(pageContainer);
}

// DOM準備完了後に実行
document.addEventListener('DOMContentLoaded', initApp);
```

## 3. ルーター詳細設計

### 3.1 Router モジュール

```javascript
// shared/router/router.js

/**
 * Hashベース軽量ルーター
 *
 * @module Router
 *
 * 公開API:
 *   Router.init(container)    - ルーター初期化
 *   Router.navigate(path)     - プログラム的な画面遷移
 *   Router.getCurrentPath()   - 現在のパス取得
 */
```

#### ルート定義テーブル

```javascript
const ROUTES = {
  '/':                  () => import('../../portal/pages/home.js'),
  '/games/typing':      () => import('../../games/typing/pages/description.js'),
  '/games/typing/play': () => import('../../games/typing/pages/play.js'),
  '/ranking':           () => import('../../user/pages/ranking.js'),
  '/login':             () => import('../../user/pages/login.js'),
  '/about':             () => import('../../portal/pages/about.js'),
  '/contact':           () => import('../../portal/pages/contact.js'),
};

// ↑ キーはhashの '#' を除去した内部パス。
// URL上は '#/games/typing' だが、Router内部では '/games/typing' として扱う。

const FALLBACK = () => import('../../portal/pages/not-found.js');
```

#### 画面遷移処理フロー

```
 hashchange イベント検知
        │
        ▼
 getPathFromHash()
   location.hash → "#/games/typing" → "/games/typing"
   hash未定義 → "/"
        │
        ▼
 ROUTES[path] 存在チェック
   ├─ 存在する → routeLoader = ROUTES[path]
   └─ 存在しない → routeLoader = FALLBACK
        │
        ▼
 現在のページを破棄
   container内のDOM除去
   前ページの destroy() 呼び出し（存在すれば）
        │
        ▼
 ローディング表示 ON
   Loading.show(container)
        │
        ▼
 動的import実行
   const module = await routeLoader()
        │
        ▼
 ローディング表示 OFF
   Loading.hide()
        │
        ▼
 新ページを描画
   const page = module.default.create()
   container.appendChild(page)
        │
        ▼
 ヘッダー/フッター表示制御
   isGamePlayPage(path)
     ├─ true  → header.hide(), footer.hide()
     └─ false → header.show(), footer.show()
        │
        ▼
 route-change イベント発火
   dispatchEvent('route-change', { path })
```

#### init メソッド詳細

```javascript
/**
 * ルーターを初期化する
 *
 * @param {HTMLElement} container - ページコンテンツを描画するコンテナ要素
 *
 * 処理:
 * 1. containerを保持
 * 2. hashchangeイベントリスナーを登録
 * 3. 現在のhashで初回ルーティングを実行
 */
static init(container) {
  this._container = container;
  this._currentPage = null;

  window.addEventListener('hashchange', () => this._onRouteChange());
  this._onRouteChange(); // 初回実行
}
```

#### navigate メソッド詳細

```javascript
/**
 * プログラム的にページ遷移する
 *
 * @param {string} path - 遷移先パス（例: '/games/typing'）
 *
 * hashを変更することでhashchangeイベントが発火し、
 * _onRouteChange()が実行される。
 */
static navigate(path) {
  window.location.hash = `#${path}`;
}
```

#### ゲームプレイ画面判定

```javascript
/**
 * ヘッダー/フッターを非表示にすべきパスかを判定
 *
 * @param {string} path
 * @returns {boolean}
 */
function isGamePlayPage(path) {
  return path.endsWith('/play');
}
```

## 4. 共通コンポーネント詳細設計

### 4.1 設計方針

- 各コンポーネントは **ES Module** として実装し、`default export` でオブジェクトを公開
- DOM生成は `document.createElement` を使用（`innerHTML` は XSS 防止のため原則不使用）
- スタイルは BEM 命名規則に従い、コンポーネントごとにCSS classを定義
- 外部データ（ユーザー入力等）をDOMに挿入する際は `textContent` を使用

```javascript
 * @property {function(): AuthUser|null} getCurrentUser        - 現在のユーザー取得
 * @property {function(): boolean} isLoggedIn                  - ログイン状態判定
 *
 * @typedef {Object} AuthResult
 * @property {boolean} success
```

#### モック実装

```javascript
// user/repository/mock-auth-repository.js

/**
 * モック認証リポジトリ
 *
 * データ永続化: sessionStorage
 *   key: 'penguin-games-auth'
 *   value: JSON.stringify({ id, name })
 *
 * login(name):
 *   1. バリデーション（1〜20文字、空白トリム後に空でないこと）
 *   2. id を 'user-' + Date.now() で生成
 *   3. sessionStorage に保存

 *

 *   1. sessionStorage からキーを削除

 *   1. sessionStorage から取得、JSON.parse

 */
```

## 8. イベント詳細設計

### 8.1 カスタムイベント一覧

| イベント名 | 発火元 | 購読先 | ペイロード |
|-----------|--------|--------|----------|
| `route-change` | Router | Header | `{ path: string }` |
| `auth-change` | MockAuthRepository | Header | `{ user: AuthUser\|null }` |
| `game-start` | ゲーム説明ページ | Router | `{ gameId: string }` |

### 8.2 イベントの発火方法

```javascript
/**
 * カスタムイベントを発火する共通ユーティリティ
 *
 * @param {string} eventName - イベント名
 * @param {Object} detail    - ペイロード
 */
function emitEvent(eventName, detail) {
  window.dispatchEvent(new CustomEvent(eventName, { detail }));
}
```

### 8.3 イベントの購読方法

```javascript
/**
 * カスタムイベントを購読する
 *
 * @param {string} eventName - イベント名
 * @param {function(CustomEvent): void} handler - ハンドラー
 * @returns {function(): void} 購読解除関数
 */
function onEvent(eventName, handler) {
  window.addEventListener(eventName, handler);
  return () => window.removeEventListener(eventName, handler);
}
```

### 8.4 イベントフロー図

```
【ページ遷移】
  ユーザー: ナビリンククリック
    → hashchange
    → Router._onRouteChange()
    → emitEvent('route-change', { path })
    → Header.setActiveNav(path)

【ログイン】
  ユーザー: ログインボタン押下
    → MockAuthRepository.login(name)
    → sessionStorage に保存
    → emitEvent('auth-change', { user: { id, name } })
    → Header: ログイン表示更新
    → Router.navigate('/')

【ログアウト】
  ユーザー: ログアウトボタン押下
    → MockAuthRepository.logout()
    → sessionStorage から削除
    → emitEvent('auth-change', { user: null })
    → Header: 未ログイン表示に戻す
```
