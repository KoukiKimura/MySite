## 2. エントリーポイント

### 2.1 main.js

アプリケーション全体の初期化を行うエントリーポイント。

```javascript
// web/src/main.js

import { Router } from './shared/router/router.js';
import { Header } from './shared/components/header.js';
import { Footer } from './shared/components/footer.js';
import { PageContainer } from './shared/components/page-container.js';
import './shared/styles/main.css';

/**
 * アプリケーション初期化
 *
 * 実行順序:
 * 1. DOM構造を構築（header → page-container → footer）
 * 2. Routerを初期化（header / footer の表示制御も委譲）
 * 3. 初回ルート解決
 */
export function initApp() {
  const app = document.getElementById('app');

  const header = Header.create();
  const pageContainer = PageContainer.create();
  const footer = Footer.create();

  app.appendChild(header);
  app.appendChild(pageContainer);
  app.appendChild(footer);

  Router.init(pageContainer, header, footer);
}

// DOM準備完了後に実行
document.addEventListener('DOMContentLoaded', initApp);
```

## 3. ルーター詳細設計

### 3.1 Router モジュール

#### ルート定義テーブル

```javascript
const ROUTES = [
  {
    pattern: /^\/$/,
    load: () => import('../../portal/pages/entry.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/(kids|teens)$/,
    load: () => import('../../portal/pages/home.js'),
    getParams: ([, mode]) => ({ mode }),
  },
  {
    pattern: /^\/(kids|teens)\/([a-z0-9-]+)$/,
    load: () => import('../../portal/pages/category.js'),
    getParams: ([, mode, category]) => ({ mode, category }),
  },
  {
    pattern: /^\/games\/typing$/,
    load: () => import('../../games/typing/pages/description.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/games\/typing\/play$/,
    load: () => import('../../games/typing/pages/play.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/ranking$/,
    load: () => import('../../user/pages/ranking.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/login$/,
    load: () => import('../../user/pages/login.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/admin$/,
    load: () => import('../../admin/pages/dashboard.js'),
    getParams: () => ({}),
    guard: 'mock-auth',
  },
  {
    pattern: /^\/parents$/,
    load: () => import('../../portal/pages/parents.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/guide$/,
    load: () => import('../../portal/pages/guide.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/safety\/links$/,
    load: () => import('../../portal/pages/safety-links.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/policy\/content$/,
    load: () => import('../../portal/pages/content-policy.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/policy\/privacy$/,
    load: () => import('../../portal/pages/privacy.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/about$/,
    load: () => import('../../portal/pages/about.js'),
    getParams: () => ({}),
  },
  {
    pattern: /^\/contact$/,
    load: () => import('../../portal/pages/contact.js'),
    getParams: () => ({}),
  },
];

const FALLBACK = () => import('../../portal/pages/not-found.js');

function resolveRoute(path) {
  for (const route of ROUTES) {
    const matched = path.match(route.pattern);
    if (matched) {
      return {
        loader: route.load,
        params: route.getParams(matched),
        guard: route.guard ?? null,
      };
    }
  }

  return {
    loader: FALLBACK,
    params: {},
    guard: null,
  };
}
```

#### 画面遷移処理フロー

```text
 hashchange イベント検知
        │
        ▼
  getPathFromHash()
   location.hash → "#/kids/games" → "/kids/games"
   hash未定義 → "/"
        │
        ▼
 resolveRoute(path)
   ├─ pattern 一致 → loader / params / guard を取得
   └─ 不一致 → FALLBACK を返す
        │
        ▼
 guard 判定
   ├─ guard = 'mock-auth' かつ未ログイン → '/login' にリダイレクト
   └─ それ以外 → 続行
        │
        ▼
  ローディング表示 ON
    Loading.show(container)
        │
        ▼
 現在のページを破棄
   container内のDOM除去
   前ページの destroy() 呼び出し（存在すれば）
        │
        ▼
  動的import実行
   const module = await loader()
        │
        ▼
  ローディング表示 OFF
   Loading.hide(container)
        │
        ▼
  新ページを描画
   const page = module.default.create({ path, params })
   container.appendChild(page)
        │
        ▼
 ヘッダー/フッター表示制御
   isGamePlayPage(path)
     ├─ true  → header.hide(), footer.hide()
     └─ false → header.show(), footer.show()
        │
        ▼
  route-change イベント発火
   dispatchEvent('route-change', { path, params })
```

#### init メソッド詳細

```javascript
/**
 * ルーターを初期化する
 *
 * @param {HTMLElement} container - ページコンテンツを描画するコンテナ要素
 * @param {HTMLElement} header - ヘッダー要素
 * @param {HTMLElement} footer - フッター要素
 *
 * 処理:
 * 1. containerを保持
 * 2. header / footer を保持
 * 2. hashchangeイベントリスナーを登録
 * 3. 現在のhashで初回ルーティングを実行
 */
static init(container, header, footer) {
  this._container = container;
  this._header = header;
  this._footer = footer;
  this._currentPage = null;

  window.addEventListener('hashchange', () => this._onRouteChange());
  this._onRouteChange(); // 初回実行
}
```

#### navigate メソッド詳細

```javascript
/**
 * プログラム的にページ遷移する
 *
 * @param {string} path - 遷移先パス（例: '/games/typing'）
 *
 * hashを変更することでhashchangeイベントが発火し、
 * _onRouteChange()が実行される。
 */
static navigate(path) {
  window.location.hash = `#${path}`;
}
```

#### ゲームプレイ画面判定

```javascript
/**
 * ヘッダー/フッターを非表示にすべきパスかを判定
 *
 * @param {string} path
 * @returns {boolean}
 */
function isGamePlayPage(path) {
  return path.endsWith('/play');
}
```

#### ガード設計

| guard | 条件 | 条件未達時 |
|-------|------|-----------|
| `mock-auth` | `mockAuthRepository.isLoggedIn() === true` | `#/login` に遷移 |

#### ルーティング補足
- `#/` は表示ページではなく共通入口であり、保存済み `siteMode` を参照して `#/kids` または `#/teens` へ遷移する
- `#/kids/:category` `#/teens/:category` はカテゴリマスタに存在しない値なら 404 とする
- `route-change` の `params` はページ側でモード / カテゴリの復元に用いる

## 4. 共通コンポーネント詳細設計

### 4.1 設計方針

- 各コンポーネントは **ES Module** として実装し、`default export` でオブジェクトを公開
- DOM生成は `document.createElement` を使用（`innerHTML` は XSS 防止のため原則不使用）
- スタイルは BEM 命名規則に従い、コンポーネントごとにCSS classを定義
- 外部データ（ユーザー入力等）をDOMに挿入する際は `textContent` を使用

#### アクティブ状態の判定ロジック

```javascript
/**
 * 現在のパスに基づきナビゲーションのアクティブ状態を更新
 *
 * @param {string} path - 現在のルートパス
 *
 * 判定ルール:
 *  - '/' '/kids' '/teens' '/kids/*' '/teens/*' → トップをアクティブ

 *  - 初期化時および 'site-mode-change' 購読時に #header__top-link / #header__mobile-top-link の

 */
```

```javascript
/**
```

## 8. イベント詳細設計

### 8.1 カスタムイベント一覧

| イベント名 | 発火元 | 購読先 | ペイロード |
|-----------|--------|--------|------------|
| `route-change` | Router | Header、PageContainer、各ページ | `{ path, params }` |
| `site-mode-change` | SiteModeState | Header、Home、Category | `{ mode, source }` |
| `auth-change` | MockAuthRepository | Header、Login、Admin | `{ user }` |
| `portal-context-update` | Home、Category | Description、ResultUI | `{ lastPortalRoute, mode, category }` |
| `search-history-update` | MockHistoryRepository | RecommendationService、Home、Category | `{ mode, category, action }` |
| `ranking-update` | MockRecordRepository | Ranking、ResultUI | `{ gameId, mode }` |
| `profile-reset` | Login | Home、Ranking | `{ scope }` |

### 8.2 イベントの発火方法

```javascript
// shared/events.js
export function emitEvent(eventName, detail) {
  window.dispatchEvent(new CustomEvent(eventName, { detail }));
}
```

### 8.3 イベントの購読方法

```javascript
// shared/events.js
export function onEvent(eventName, handler) {
  window.addEventListener(eventName, handler);
  return () => window.removeEventListener(eventName, handler);
}
```

### 8.4 イベントフロー図

```
【ページ遷移】
  ユーザー: ナビゲーション操作
    -> Router.resolveRoute(path)
    -> emitEvent('route-change', { path, params })
    -> Header / PageContainer が表示更新

【サイト切替】
  ユーザー: ヘッダー or ヒーローの切替操作
    -> setSiteMode(mode)
    -> localStorage 保存
    -> emitEvent('site-mode-change', { mode, source })
    -> Router.navigate(`/${mode}`)

【カテゴリ遷移】
  ユーザー: カテゴリレールクリック
    -> Router.navigate(`/${mode}/${category}`)
    -> emitEvent('portal-context-update', { lastPortalRoute, mode, category })

【検索履歴保存】
  ユーザー: 検索送信 or キーワードクリック
    -> MockHistoryRepository.save(...)
    -> localStorage 保存
    -> emitEvent('search-history-update', { mode, category, action: 'save' })

【ゲーム終了】
  ゲーム: リザルト確定
    -> MockRecordRepository.save(record)
    -> emitEvent('ranking-update', { gameId: 'typing', mode })
```

## 10. エラーハンドリング設計

### 10.1 エラー分類

| カテゴリ | 例 | 対処 |
|---------|-----|------|
| ルーティングエラー | 未定義パス、未定義カテゴリ | 404ページへフォールバック |

| モジュールロードエラー | 動的importの失敗 | 404ページにフォールバック + console.error |

## 11. セキュリティ実装詳細

### 11.1 フロントエンド対策

| 対策 | 実装方法 |
|------|----------|
| XSS対策 | 文字列は `textContent` で描画し、ユーザー入力を `innerHTML` に入れない |
| URL制御 | 遷移先は定義済みルートのみ許可し、検索語を URL に埋め込まない |
| 管理画面保護 | `mock-auth` ガードで未ログイン遷移を抑止 |
| iframe 制御 | Google Forms に `sandbox` を付与 |
| ローカルデータ | 生パスワードを保持しない |

### 11.2 将来の外部検索フィルタ方針

| モード | 方針 |
|--------|------|
| キッズ | 性的表現、過度な暴力、危険行為、SNS強導線、詐欺 / ウイルス誘導を除外 |
| ティーンズβ | 一般ニュースや音楽は許容しつつ、性的表現、過度な暴力、詐欺 / ウイルス誘導は除外 |
