### 2.1 Vector2

```javascript
// domain/values/vector2.js

/**
 * 2次元ベクトルを表す値オブジェクト（不変）
 * 位置・速度・法線ベクトルなどに使用する。
 */

export function createVector2(x, y) {
  return Object.freeze({ x, y });
}

export const Vector2 = {
  create: (x, y) => createVector2(x, y),
  add: (a, b) => createVector2(a.x + b.x, a.y + b.y),
  sub: (a, b) => createVector2(a.x - b.x, a.y - b.y),
  scale: (v, s) => createVector2(v.x * s, v.y * s),
  magnitude: (v) => Math.sqrt(v.x * v.x + v.y * v.y),
  normalize: (v) => {
    const mag = Vector2.magnitude(v);
    if (mag === 0) return createVector2(0, 0);
    return createVector2(v.x / mag, v.y / mag);
  },
  /**
   * 法線ベクトルに基づく反射ベクトルを返す
   * reflect = v - 2 * (v・n) * n
   * @param {Vector2} v       - 入射ベクトル
   * @param {Vector2} normal  - 正規化された法線ベクトル
   * @returns {Vector2}
   */
  reflect: (v, normal) => {
    const dot = v.x * normal.x + v.y * normal.y;
    return createVector2(
      v.x - 2 * dot * normal.x,
      v.y - 2 * dot * normal.y,
    );
  },
};
```

```javascript
/**
 * ヒット処理（hp減算・破壊判定）
 * INVINCIBLE（hp=null）の場合は変化なし
 * @param {Block} block
 * @returns {{ block: Block, destroyed: boolean }}
 */
export function hitBlock(block) {
  if (block.hp === null) {
    return { block, destroyed: false };
  }
  const newHp = block.hp - 1;
  if (newHp <= 0) {
    return { block: createBlock({ ...block, hp: 0, isAlive: false }), destroyed: true };
  }
  return { block: createBlock({ ...block, hp: newHp }), destroyed: false };
}
```

### 4.1 CollisionDetector

```javascript
// domain/services/collision-detector.js

const FIELD_WIDTH  = 960;
const FIELD_HEIGHT = 640;

/**
 * ボールと壁の衝突判定
 * @param {Ball} ball
 * @returns {{ hitLeft: boolean, hitRight: boolean, hitTop: boolean, hitBottom: boolean }}
 */
export function checkBallWall(ball) {
  return {
    hitLeft:   ball.position.x - ball.radius <= 0,
    hitRight:  ball.position.x + ball.radius >= FIELD_WIDTH,
    hitTop:    ball.position.y - ball.radius <= 0,
    hitBottom: ball.position.y + ball.radius >= FIELD_HEIGHT,
  };
}

/**
 * ボールとパドルの衝突判定
 * @param {Ball} ball
 * @param {Paddle} paddle
 * @returns {{ hit: boolean, reflectVelocity: Vector2|null }}
 *
 * 反射角の計算:
 *   パドル中央からの相対位置 offset = (ballX - paddleCenterX) / (paddleWidth / 2)  [-1, 1]
 *   反射角 = offset × MAX_REFLECT_ANGLE (最大75°)
 *   速度: { x: speed × sin(angle), y: -speed × cos(angle) }
 */
export function checkBallPaddle(ball, paddle) {
  const paddleCenterX = paddle.x + paddle.width / 2;
  const paddleTop     = paddle.y;

  // ボールがパドルの矩形内に入っているか
  const inX = ball.position.x + ball.radius > paddle.x
           && ball.position.x - ball.radius < paddle.x + paddle.width;
  const inY = ball.position.y + ball.radius >= paddleTop
           && ball.position.y - ball.radius <= paddleTop + paddle.height;
  // 上方向から衝突しているか（下方向の速度を持っているとき）
  const movingDown = ball.velocity.y > 0;

  if (!inX || !inY || !movingDown) {
    return { hit: false, reflectVelocity: null };
  }

  const MAX_REFLECT_ANGLE = Math.PI * 5 / 12; // 75°
  const offset = (ball.position.x - paddleCenterX) / (paddle.width / 2);
  const angle  = offset * MAX_REFLECT_ANGLE;
  const speed  = Vector2.magnitude(ball.velocity);

  return {
    hit: true,
    reflectVelocity: Vector2.create(
      speed * Math.sin(angle),
      -speed * Math.abs(Math.cos(angle)),
    ),
  };
}

/**
 * ボールとブロックの衝突判定（円 vs 矩形）
 * @param {Ball} ball
 * @param {{ x, y, width, height }} blockRect
 * @returns {{ hit: boolean, normal: Vector2|null }}
 */
export function checkBallBlock(ball, blockRect) {
  const closestX = Math.max(blockRect.x, Math.min(ball.position.x, blockRect.x + blockRect.width));
  const closestY = Math.max(blockRect.y, Math.min(ball.position.y, blockRect.y + blockRect.height));
  const dx = ball.position.x - closestX;
  const dy = ball.position.y - closestY;
  const distSq = dx * dx + dy * dy;

  if (distSq > ball.radius * ball.radius) {
    return { hit: false, normal: null };
  }

  // 衝突法線を最近傍辺から求める
  const overlapLeft   = (ball.position.x + ball.radius) - blockRect.x;
  const overlapRight  = (blockRect.x + blockRect.width) - (ball.position.x - ball.radius);
  const overlapTop    = (ball.position.y + ball.radius) - blockRect.y;
  const overlapBottom = (blockRect.y + blockRect.height) - (ball.position.y - ball.radius);
  const minOverlap    = Math.min(overlapLeft, overlapRight, overlapTop, overlapBottom);

  let normal;
  if (minOverlap === overlapLeft)        normal = Vector2.create(-1, 0);
  else if (minOverlap === overlapRight)  normal = Vector2.create(1, 0);
  else if (minOverlap === overlapTop)    normal = Vector2.create(0, -1);
  else                                   normal = Vector2.create(0, 1);

  return { hit: true, normal };
}

/**
 * 落下アイテムとパドルの衝突判定（矩形 vs 矩形）
 * @param {PowerUpItem} item
 * @param {Paddle} paddle
 * @returns {boolean}
 */
export function checkItemPaddle(item, paddle) {
  return item.position.x < paddle.x + paddle.width
      && item.position.x + item.width > paddle.x
      && item.position.y < paddle.y + paddle.height
      && item.position.y + item.height > paddle.y;
}
```

### 4.2 PhysicsEngine

```javascript
// domain/services/physics-engine.js

const FALL_SPEED = 2; // パワーアップアイテムの落下速度 px/frame

/**
 * ボールを deltaTime 分移動させる（60fps基準換算）
 * @param {Ball} ball
 * @param {number} deltaTime - ms
 * @returns {Ball}
 */
export function updateBallPosition(ball, deltaTime) {
  const factor = deltaTime / (1000 / 60); // 60fps基準に正規化
  const newPos = Vector2.add(ball.position, Vector2.scale(ball.velocity, factor));
  return updateBall(ball, { position: newPos });
}

/**
 * パワーアップアイテムを1フレーム分落下させる
 * @param {PowerUpItem} item
 * @returns {PowerUpItem}
 */
export function updateItemFall(item) {
  return {
    ...item,
    position: Vector2.create(item.position.x, item.position.y + FALL_SPEED),
  };
}

/**
 * 速度に倍率を適用する（SLOW効果等）
 * @param {Ball} ball
 * @param {number} multiplier
 * @returns {Ball}
 */
export function applySpeedMultiplier(ball, multiplier) {
  return updateBall(ball, { velocity: Vector2.scale(ball.velocity, multiplier) });
}
```

### 5.1 UC-1: StartGame

```javascript
// application/use-cases/start-game.js

/**
 * ゲーム開始処理
 *
 * 1. StageRepository からステージデータを取得
 * 2. Difficulty に基づいて初期 Paddle / Ball を生成
 * 3. GameSession を初期化
 * 4. カウントダウン開始のコールバックを呼ぶ
 *
 * @param {Object} params
 * @param {string} params.difficulty         - Difficulty.id
 * @param {StageRepository} params.stageRepo
 * @param {Function} params.onReady          - (session: GameSession) => void
 */
export async function startGame({ difficulty, stageRepo, onReady }) {
  const stagesData = await stageRepo.fetchByDifficulty(difficulty);
  const stages     = stagesData.map(StageBuilder.build);
  const diff       = getDifficultyById(difficulty);
  const paddle     = initPaddle(diff);
  const ball       = initBall(paddle);
  const session    = createGameSession({ difficulty, stages, paddle, initialBall: ball });
  onReady(session);
}
```

### 5.2 UC-2: UpdateFrame

```javascript
// application/use-cases/update-frame.js

/**
 * 1フレーム分の更新処理
 *
 * @param {GameSession} session
 * @param {number}      deltaTime - ms
 * @returns {GameSession} 更新後のセッション（不変オブジェクトではなく変異可。詳細は実装判断）
 */
export function updateFrame(session, deltaTime) {
  if (session.state !== SessionState.PLAYING) return session;

  let { balls, blocks, powerUpItems, score, combo, maxCombo, lives, activeEffects } = session;

  // 1. ボール物理更新 + 衝突処理
  const slowMultiplier = activeEffects.includes('slow') ? 0.5 : 1.0;
  balls = balls.map(ball => {
    ball = updateBallPosition(ball, deltaTime);  // PhysicsEngine

    // 壁衝突
    const wallHit = checkBallWall(ball);         // CollisionDetector
    if (wallHit.hitLeft || wallHit.hitRight) {
      ball = updateBallVelocity(ball, Vector2.create(-ball.velocity.x, ball.velocity.y));
    }
    if (wallHit.hitTop) {
      ball = updateBallVelocity(ball, Vector2.create(ball.velocity.x, -ball.velocity.y));
    }
    if (wallHit.hitBottom) {
      ball = updateBallState(ball, BallState.LOST);
      return ball;
    }

    // パドル衝突
    const paddleResult = checkBallPaddle(ball, session.paddle);
    if (paddleResult.hit) {
      ball = updateBallVelocity(ball, paddleResult.reflectVelocity);
    }

    // ブロック衝突
    for (const row of blocks) {
      for (let i = 0; i < row.length; i++) {
        const block = row[i];
        if (!block || !block.isAlive) continue;
        const rect  = getBlockRect(block);
        const { hit, normal } = checkBallBlock(ball, rect);
        if (hit) {
          ball = updateBallVelocity(ball, Vector2.reflect(ball.velocity, normal));
          const { block: newBlock, destroyed } = hitBlock(block);
          row[i] = newBlock;
          if (destroyed) {
            const blockType = getBlockTypeById(newBlock.type);
            score += calcBlockScore(blockType, combo);
            combo++;
            maxCombo = Math.max(maxCombo, combo);
            if (newBlock.type === 'power-up') {
              powerUpItems = [...powerUpItems, spawnPowerUpItem(rect)];
            }
          }
          break; // 1フレームで1ブロックまで（多重衝突防止）
        }
      }
    }
    return ball;
  });

  // 2. ボール消滅チェック → LoseLife or ミスなし
  const activeBalls = balls.filter(b => b.state !== BallState.LOST);
  if (activeBalls.length === 0) {
    // LoseLife ユースケースは呼び出し元が dispatch
    return { ...session, balls, blocks, powerUpItems, score, combo: 0, maxCombo,
             state: SessionState.LOST_LIFE };
  }

  // 3. パワーアップアイテム落下 + 取得判定
  powerUpItems = powerUpItems
    .map(item => updateItemFall(item))
    .filter(item => {
      if (checkItemPaddle(item, session.paddle)) {
        score += getPowerUpBonus();
        // applyPowerUp は呼び出し元が dispatch
        return false;
      }
      return item.position.y < FIELD_HEIGHT;
    });

  // 4. 全ブロック破壊チェック
  const alive = blocks.flat().filter(b => b && b.isAlive && b.type !== 'invincible');
  if (alive.length === 0) {
    return { ...session, balls: activeBalls, blocks, powerUpItems, score, combo, maxCombo,
             state: SessionState.STAGE_CLEAR };
  }

  return { ...session, balls: activeBalls, blocks, powerUpItems, score, combo, maxCombo, activeEffects };
}
```

```javascript
/**
```

```javascript
 *
 * @param {HTMLCanvasElement} canvas

  canvas.height = BASE_HEIGHT;
  const containerWidth = canvas.parentElement?.clientWidth ?? BASE_WIDTH;
  const scale = Math.min(1, containerWidth / BASE_WIDTH);

  canvas.style.transformOrigin = 'top left';
  canvas.style.width  = `${BASE_WIDTH}px`;
  canvas.style.height = `${BASE_HEIGHT}px`;
  const ctx = canvas.getContext('2d');
  return { ctx, scale };
}
```

### 6.2 ゲームループ

```javascript
// presentation/canvas/game-loop.js

/**
 * requestAnimationFrame を使ったゲームループ
 *
 * @param {Function} updateFn - (deltaTime: number) => void
 * @param {Function} renderFn - () => void
 * @returns {{ start: Function, stop: Function }}
 */
export function createGameLoop(updateFn, renderFn) {
  let rafId     = null;
  let lastTime  = null;
  const MAX_DELTA = 50; // 最大deltaTime（フレームスキップ防止）

  function loop(timestamp) {
    if (lastTime === null) lastTime = timestamp;
    const deltaTime = Math.min(timestamp - lastTime, MAX_DELTA);
    lastTime = timestamp;
    updateFn(deltaTime);
    renderFn();
    rafId = requestAnimationFrame(loop);
  }

  return {
    start: () => { lastTime = null; rafId = requestAnimationFrame(loop); },
    stop:  () => { if (rafId) cancelAnimationFrame(rafId); rafId = null; },
  };
}
```

### 6.3 入力ハンドラ

```javascript
// presentation/input/mouse-handler.js

/**
 * マウス移動でパドルを操作する
 *
 * @param {HTMLCanvasElement} canvas
 * @param {Function} onMove - (x: number) => void  ※ Canvas座標系（スケール補正済み）
 * @param {number} scale    - Canvas の CSS スケール係数
 * @returns {{ attach: Function, detach: Function }}
 */
export function createMouseHandler(canvas, onMove, scale) {
  const handler = (e) => {
    const rect = canvas.getBoundingClientRect();
    const x    = (e.clientX - rect.left) / scale;
    onMove(x);
  };
  return {
    attach: () => canvas.addEventListener('mousemove', handler),
    detach: () => canvas.removeEventListener('mousemove', handler),
  };
}
```

```javascript
 * タッチ操作でパドルを操作する
 * マルチタッチは最初のタッチ点のみを使用する。

export function createTouchHandler(canvas, onMove, scale) {
  const handler = (e) => {
    e.preventDefault();
    const touch = e.touches[0];
    const rect  = canvas.getBoundingClientRect();
    const x     = (touch.clientX - rect.left) / scale;
    onMove(x);
  };
  return {
    attach: () => canvas.addEventListener('touchmove', handler, { passive: false }),
    detach: () => canvas.removeEventListener('touchmove', handler),
  };
}
```

```javascript
const PADDLE_SPEED = 8; // px/frame（キーボード操作時の移動速度）

/**
 * ←→キーによるパドル操作とSpace/P/Escapeのゲーム操作
 */
export function createKeyboardHandler({ onPaddleMove, onLaunch, onPause }) {
  const keys = new Set();

  const keydown = (e) => {
    keys.add(e.key);
    if (e.key === ' ')                     onLaunch?.();
    if (e.key === 'p' || e.key === 'Escape') onPause?.();
  };
  const keyup = (e) => keys.delete(e.key);

  /** ゲームループから毎フレーム呼ぶ */
  function tick() {
    if (keys.has('ArrowLeft'))  onPaddleMove?.(-PADDLE_SPEED);
    if (keys.has('ArrowRight')) onPaddleMove?.(+PADDLE_SPEED);
  }

  return {
    attach: () => {
      window.addEventListener('keydown', keydown);
      window.addEventListener('keyup',   keyup);
    },
    detach: () => {
      window.removeEventListener('keydown', keydown);
      window.removeEventListener('keyup',   keyup);
    },
    tick,
  };
}
```

```javascript
// infrastructure/repositories/json-stage-repository.js

import { StageRepository } from '../../application/repositories/stage-repository.js';

/**
 * JSON ファイルからステージデータを取得するリポジトリ実装
 *
 * データファイルパス: /public/data/stages-{difficulty}.json
 */
export class JsonStageRepository extends StageRepository {
  #cache = new Map();

  async fetchByDifficulty(difficulty) {
    if (this.#cache.has(difficulty)) {
      return this.#cache.get(difficulty);
    }
    const res    = await fetch(`/data/stages-${difficulty}.json`);
    if (!res.ok) throw new Error(`Failed to load stages: ${difficulty}`);
    const json   = await res.json();
    const stages = json.stages;
    this.#cache.set(difficulty, stages);
    return stages;
  }
}
```
