```javascript
// domain/values/game-config.js

/**
 * ゲーム設定の組み合わせ
 *
 * 不変の値オブジェクト。メニュー画面でユーザーが選択した設定を保持する。
 *
 * @typedef {Object} GameConfig
 * @property {string} gameMode      - GameMode の値
 * @property {string} difficulty    - Difficulty の id
 * @property {string} questionMode  - QuestionMode の値
 * @property {number} timeLimit     - 制限時間（秒）※ TIME_LIMIT モード時
 * @property {number} questionCount - 問題数 ※ FIXED_COUNT モード時
 */

/** デフォルト設定テーブル */
const CONFIG_TABLE = {
  'time-limit': {
    easy:   { timeLimit: 90,  questionCount: 0 },
    normal: { timeLimit: 60,  questionCount: 0 },
    hard:   { timeLimit: 60,  questionCount: 0 },  // 基本設計書の「60〜90秒」範囲に準拠
  },
  'fixed-count': {
    easy:   { timeLimit: 0, questionCount: 10 },
    normal: { timeLimit: 0, questionCount: 15 },
    hard:   { timeLimit: 0, questionCount: 20 },
  },
};

/**
 * ゲーム設定を生成する
 *
 * @param {string} gameMode - GameMode値
 * @param {string} difficulty - Difficulty id
 * @param {string} questionMode - QuestionMode値
 * @returns {GameConfig}
 * @throws {Error} 不正な組み合わせの場合
 */
export function createGameConfig(gameMode, difficulty, questionMode) {
  const modeConfig = CONFIG_TABLE[gameMode];
  if (!modeConfig) throw new Error(`Invalid gameMode: ${gameMode}`);
  const diffConfig = modeConfig[difficulty];
  if (!diffConfig) throw new Error(`Invalid difficulty: ${difficulty}`);

  return Object.freeze({
    gameMode,
    difficulty,
    questionMode,
    timeLimit: diffConfig.timeLimit,
    questionCount: diffConfig.questionCount,
  });
}
```

 *
 * @typedef {Object} Score

 * @property {number} comboBonus   - コンボボーナス合計

 * スコアを生成する
 *
 * @param {number} baseScore
 * @param {number} comboBonus
 * @param {number} missPenalty
 * @returns {Score}
 */
export function createScore(baseScore, comboBonus, missPenalty) {
  const total = Math.max(0, baseScore + comboBonus - missPenalty);
  return Object.freeze({ baseScore, comboBonus, missPenalty, total });
}

/**
 * 初期スコア（ゲーム開始時）
 * @returns {Score}
 */
export function initialScore() {
  return createScore(0, 0, 0);
}

### 2.6 RomajiInput

```javascript
// domain/values/romaji-input.js

/**
 * 1つのかなチャンクに対するローマ字入力状態
 *
 * @typedef {Object} RomajiInput
 * @property {string} kana             - 対象のかな（例: "しゃ"）
 * @property {string[]} candidates     - 残りの候補パターン
 * @property {number} matchedLength    - 入力済みの文字数
 * @property {boolean} isComplete      - このチャンクの入力が完了したか
 */

/**
 * RomajiInput を生成
 *
 * @param {string} kana - 対象のかな
 * @param {string[]} patterns - 許容されるローマ字パターンの配列
 * @returns {RomajiInput}
 */
export function createRomajiInput(kana, patterns) {
  return Object.freeze({
    kana,
    candidates: [...patterns],
    matchedLength: 0,
    isComplete: false,
  });
}

/**
 * 1文字入力を受けて次の状態を返す
 *
 * @param {RomajiInput} input - 現在の入力状態
 * @param {string} char - 入力された1文字（小文字英字）
 * @returns {{ next: RomajiInput|null, isHit: boolean }}
 *   - next: 次の状態（null=ミス）
 *   - isHit: 正解の入力だったか
 *
 * アルゴリズム:
 * 1. 候補パターンのうち、matchedLength番目の文字が char と一致するものを残す
 * 2. 候補が0 → ミス（{ next: null, isHit: false }）
 * 3. 残った候補のいずれかが matchedLength+1 で全文字一致 → 完了
 * 4. それ以外 → matchedLength+1 で継続
 */
export function advanceRomajiInput(input, char) {
  const nextPos = input.matchedLength;
  const remaining = input.candidates.filter(p => p[nextPos] === char);

  if (remaining.length === 0) {
    return { next: null, isHit: false };
  }

  const newLength = nextPos + 1;
  const isComplete = remaining.some(p => p.length === newLength);

  return {
    next: Object.freeze({
      kana: input.kana,
      candidates: remaining,
      matchedLength: newLength,
      isComplete,
    }),
    isHit: true,
  };
}
```

#### 状態遷移メソッド

```javascript
/**
 * ゲームを開始する
 * @param {GameSession} session
 * @returns {GameSession}
 * @throws {Error} state が READY でない場合
 */
export function startSession(session) {
  if (session.state !== SessionState.READY) {
    throw new Error(`Cannot start: current state is ${session.state}`);
  }
  session.state = SessionState.PLAYING;
  session.startTime = Date.now();
  return session;
}

/**
 * ゲームを一時停止する
 * @param {GameSession} session
 * @returns {GameSession}
 * @throws {Error} state が PLAYING でない場合
 */
export function pauseSession(session) {
  if (session.state !== SessionState.PLAYING) {
    throw new Error(`Cannot pause: current state is ${session.state}`);
  }
  session.state = SessionState.PAUSED;
  session._pauseStartTime = Date.now();
  return session;
}

/**
 * ゲームを再開する
 * @param {GameSession} session
 * @returns {GameSession}
 * @throws {Error} state が PAUSED でない場合
 */
export function resumeSession(session) {
  if (session.state !== SessionState.PAUSED) {
    throw new Error(`Cannot resume: current state is ${session.state}`);
  }
  session.pausedDuration += Date.now() - session._pauseStartTime;
  session._pauseStartTime = 0;
  session.state = SessionState.PLAYING;
  return session;
}

/**
 * ゲームを終了する
 * @param {GameSession} session
 * @returns {GameSession}
 * @throws {Error} state が PLAYING でない場合
 */
export function finishSession(session) {
  if (session.state !== SessionState.PLAYING) {
    throw new Error(`Cannot finish: current state is ${session.state}`);
  }
  session.state = SessionState.FINISHED;
  session.endTime = Date.now();
  session.elapsedTime = session.endTime - session.startTime - session.pausedDuration;
  return session;
}
```

  session.totalKeystrokes += 1;
  session.combo = 0;
  return session;
}

/**
 * 次の問題に進む
 * @param {GameSession} session
 * @returns {GameSession}

  session.currentIndex += 1;
  return session;
}

/**
 * 現在の問題を取得
 * @param {GameSession} session
 * @returns {Question|null} 全問完了ならnull
 */
export function getCurrentQuestion(session) {
  if (session.currentIndex >= session.questions.length) return null;
  return session.questions[session.currentIndex];
}

/**
 * ゲーム終了条件をチェック
 * @param {GameSession} session
 * @returns {boolean}
 *
 * 判定ルール:
 * - FIXED_COUNT: currentIndex >= questions.length
 * - TIME_LIMIT:  経過時間 >= config.timeLimit * 1000
 */
export function shouldFinish(session) {
  if (session.config.gameMode === 'fixed-count') {
    return session.currentIndex >= session.questions.length;
  }
  // time-limit
  const elapsed = Date.now() - session.startTime - session.pausedDuration;
  return elapsed >= session.config.timeLimit * 1000;
}

#### かなチャンク分解アルゴリズム

```javascript
/**
 * ひらがな文字列をかなチャンクに分解
 *
 * @param {string} reading - ひらがな文字列（例: "でんしゃ"）
 * @returns {string[]} かなチャンクの配列（例: ["で", "ん", "しゃ"]）
 *
 * 分解ルール（優先順位順）:
 * 1. 拗音（2文字）: 次の文字が 'ゃ','ゅ','ょ' なら2文字で1チャンク
 * 2. 促音 'っ': 単独で1チャンク
 * 3. 撥音 'ん': 単独で1チャンク
 * 4. その他: 1文字で1チャンク
 *
 * 例:
 *  "でんしゃ" → ["で", "ん", "しゃ"]
 *  "がっこう"  → ["が", "っ", "こ", "う"]
 *  "ちゅうしゃじょう" → ["ちゅ", "う", "しゃ", "じょ", "う"]
 */
export function decompose(reading) {
  const chunks = [];
  let i = 0;
  while (i < reading.length) {
    // 拗音チェック: 次の文字が小文字ゃゅょの場合
    if (i + 1 < reading.length && 'ゃゅょ'.includes(reading[i + 1])) {
      chunks.push(reading.substring(i, i + 2));
      i += 2;
    } else {
      chunks.push(reading[i]);
      i += 1;
    }
  }
  return chunks;
}
```

#### 「ん」の処理

```javascript
/**
 * 「ん」のローマ字パターンを返す
 *
 * 「ん」のパターンは次のチャンクに依存する:
 *
 * @param {string|null} nextChunk - 次のかなチャンク（末尾ならnull）
 * @returns {string[]}
 *
 * ルール:
 * 1. 次が null（文末）       → ['nn']
 * 2. 次が母音(あいうえお)     → ['nn']
 * 3. 次が 'な','に','ぬ','ね','の' 行  → ['nn']
 * 4. 次が 'や','ゆ','よ'              → ['nn']
 * 5. 上記以外（次が子音始まり）       → ['n', 'nn']
 *    ※ 次チャンクのローマ字パターンが子音で始まる場合、'n' 単独で確定可能。
 *    ※ 「ん」+ま行（まみむめも）の場合: 'n' + 'm...' で確定できるため
 *      ['n', 'nn'] を返す。例: 「んま」→ "nma" または "nnma"
 */
export function getNPatterns(nextChunk) {
  if (nextChunk === null) {
    return ['nn'];
  }
  const firstChar = nextChunk[0];
  const vowels = 'あいうえお';
  const naRow = 'なにぬねの';
  const yaRow = 'やゆよ';

  if (vowels.includes(firstChar) || naRow.includes(firstChar) || yaRow.includes(firstChar)) {
    return ['nn'];
  }
  return ['n', 'nn'];
}
```

#### 「っ」(促音)の処理

```javascript
/**
 * 「っ」のローマ字パターンを返す
 *
 * @param {string|null} nextChunk - 次のかなチャンク
 * @returns {string[]}
 *
 * ルール:
 * 1. 次のチャンクのパターンの先頭子音を重ねる
 *    例: 次が "か" → ["kk"]のプレフィックス（実際には次のチャンクと結合）
 * 2. 追加パターン: ["xtu", "xtsu"]
 *
 * ※ 実装方針:
 *   「っ」+次チャンクを1つのまとまりとして処理する。
 *   次チャンクの全パターンの先頭子音を重ねたパターンを生成。
 *
 * 例: っち → ["tti", "cchi", "tchi"]
 *     っか → ["kka"]
 *     っし → ["ssi", "sshi", "cci"]
 */
export function getTsuPatterns(nextChunk) {
  if (!nextChunk) return ['xtu', 'xtsu'];

  const nextPatterns = ROMAJI_TABLE[nextChunk] || [];
  const combined = [];

  for (const pattern of nextPatterns) {
    // 先頭の子音を重ねる
    combined.push(pattern[0] + pattern);
  }
  // xtu/xtsu による入力も許容
  for (const pattern of nextPatterns) {
    combined.push('xtu' + pattern);
    combined.push('xtsu' + pattern);
  }

  return combined;
}
```

> **xtu/xtsu パスの照合方針**:
> `buildMatchData` は「っ」+次チャンクを結合し、`getTsuPatterns` が返す全パターン（子音重ね・xtu・xtsu）を
> 1つのチャンクの candidates として `RomajiInput` に渡す。
> 例: 「っこ」→ candidates = `["kko", "xtuko", "xtsuko"]`
>
> `advanceRomajiInput` は候補の先頭から1文字ずつ照合するため:
> - `k` 入力 → `["kko"]` に絞られ、次に `k` → `o` で完了
> - `x` 入力 → `["xtuko", "xtsuko"]` に絞られ、以降 `t→u→k→o` または `t→s→u→k→o` で完了
>
> xtu/xtsu パスでは「っ」部分と次チャンク部分が1つの候補文字列内に連結されているため、
> チャンク間の遷移は不要。`RomajiInput` の候補絞り込みロジックだけで正しく処理される。

#### 照合用データ構造の生成

```javascript
/**
 * ひらがな文字列から照合用データ構造を生成
 *
 * @param {string} reading - ひらがな文字列
 * @returns {Array<{ kana: string, patterns: string[] }>}
 *
 * 例: "でんしゃ"
 * [
 *   { kana: "で",  patterns: ["de"] },
 *   { kana: "ん",  patterns: ["n", "nn"] },
 *   { kana: "しゃ", patterns: ["sya", "sha"] },
 * ]
 *
 * 例: "がっこう"
 * → 促音「っ」+「こ」を結合
 * [
 *   { kana: "が",    patterns: ["ga"] },
 *   { kana: "っこ",  patterns: ["kko", "xtuko", "xtsuko"] },
 *   { kana: "う",    patterns: ["u"] },
 * ]
 */
export function buildMatchData(reading) {
  const chunks = decompose(reading);
  const result = [];

  for (let i = 0; i < chunks.length; i++) {
    const chunk = chunks[i];
    const nextChunk = i + 1 < chunks.length ? chunks[i + 1] : null;

    if (chunk === 'ん') {
      result.push({ kana: chunk, patterns: getNPatterns(nextChunk) });
    } else if (chunk === 'っ') {
      // 促音+次チャンクを結合
      if (nextChunk) {
        result.push({ kana: chunk + nextChunk, patterns: getTsuPatterns(nextChunk) });
        i++; // 次のチャンクをスキップ
      } else {
        result.push({ kana: chunk, patterns: ['xtu', 'xtsu'] });
      }
    } else {
      const patterns = ROMAJI_TABLE[chunk];
      if (patterns) {
        result.push({ kana: chunk, patterns });
      }
    }
  }
  return result;
}
```

#### コンボ倍率テーブルと計算

```javascript
/**
 * コンボ倍率テーブル
 *
 * 適用単位: キーストロークごと
 * 正タイプ1打鍵ごとに 10 × その時点のコンボ倍率 を加算
 */
const COMBO_TABLE = [
  { min: 20, multiplier: 3.0 },
  { min: 10, multiplier: 2.0 },
  { min: 5,  multiplier: 1.5 },
  { min: 0,  multiplier: 1.0 },
];

/**
 * コンボ数から倍率を取得
 * @param {number} combo
 * @returns {number}
 */
export function getComboMultiplier(combo) {
  for (const entry of COMBO_TABLE) {
    if (combo >= entry.min) return entry.multiplier;

  return 1.0;

#### 最終スコア計算

```javascript
/**
 * ゲーム終了時の最終スコアを算出
 *
 * @param {GameSession} session
 * @returns {Object} 結果オブジェクト
 *
 * 計算フロー:
 * 1. baseScore = comboHistory の各打鍵スコアを合算
 *      各打鍵: 10 × getComboMultiplier(その時点のcombo)
 * 2. missPenalty = missCount × 5
 * 3. total = max(0, baseScore - missPenalty)
 * 4. wpm = calculateWPM(correctCount, elapsedTime)
 * 5. accuracy = calculateAccuracy(correctCount, totalKeystrokes)
 * 6. rank = getRank(accuracy, wpm)
 */
export function calculateFinalResult(session) {
  // 各正タイプ打鍵のスコアを合算
  let baseScore = 0;
  for (const comboAtHit of session.comboHistory) {
    baseScore += Math.round(10 * getComboMultiplier(comboAtHit));
  }

  const missPenalty = session.missCount * 5;
  const total = Math.max(0, baseScore - missPenalty);
  const wpm = calculateWPM(session.correctCount, session.elapsedTime);
  const accuracy = calculateAccuracy(session.correctCount, session.totalKeystrokes);
  const rank = getRank(accuracy, wpm);

  return Object.freeze({
    baseScore,
    comboBonus: baseScore - (session.correctCount * 10),  // 倍率分のみ
    missPenalty,
    total,
    wpm,
    accuracy,
    rank,
    correctCount: session.correctCount,
    missCount: session.missCount,
    maxCombo: session.maxCombo,
    elapsedTime: session.elapsedTime,
  });
}

#### WPM・正確率・ランク

```javascript
/**
 * WPM (Words Per Minute) を算出
 * @param {number} correctCount - 正解タイプ数
 * @param {number} ms - 経過時間（ミリ秒）
 * @returns {number} WPM（小数第1位まで）
 */
export function calculateWPM(correctCount, ms) {
  if (ms <= 0) return 0;
  const minutes = ms / 60000;
  return Math.round(((correctCount / 5) / minutes) * 10) / 10;
}

/**
 * 正確率を算出
 * @param {number} correct - 正解タイプ数
 * @param {number} total - 総キーストローク数
 * @returns {number} 正確率（%、小数第1位まで）
 */
export function calculateAccuracy(correct, total) {
  if (total === 0) return 0;
  return Math.round((correct / total) * 1000) / 10;
}

/**
 * ランクを判定
 * @param {number} accuracy - 正確率 (%)
 * @param {number} wpm - WPM
 * @returns {string} 'S' | 'A' | 'B' | 'C' | 'D'
 */
export function getRank(accuracy, wpm) {
  if (accuracy >= 98 && wpm >= 80) return 'S';
  if (accuracy >= 95 && wpm >= 60) return 'A';
  if (accuracy >= 90 && wpm >= 40) return 'B';
  if (accuracy >= 80 && wpm >= 25) return 'C';
  return 'D';
}

/**
 * 配列をシャッフル（破壊的）
 * Fisher-Yatesアルゴリズム
 *
 * @param {Array} array
 * @returns {Array} シャッフル後の同一配列
 */
function shuffle(array) {
  for (let i = array.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [array[i], array[j]] = [array[j], array[i]];
  }

}

/**
 * JSON出題データリポジトリ
 *
 * QuestionRepository インターフェースの実装。
 * infrastructure/data/ 配下の JSON ファイルを fetch で取得する。
 *
 * キャッシュ: 同一ファイルの2回目以降の取得はメモリキャッシュを返す
 */

/**
 * @returns {QuestionRepository}
 */
export function createJsonQuestionRepository() {
  const cache = new Map();

  return {
    /**
     * @param {string} questionMode
     * @param {string} difficulty
     * @returns {Promise<Object[]>}
     */
    async fetchQuestions(questionMode, difficulty) {
      const fileName = `${questionMode}-${difficulty}.json`;
      if (cache.has(fileName)) {
        return cache.get(fileName);
      }

      const url = new URL(
        `../data/${fileName}`,
        import.meta.url
      );
      const response = await fetch(url);
      if (!response.ok) {
        throw new Error(`出題データの取得に失敗: ${fileName} (${response.status})`);
      }
      const data = await response.json();
      const questions = data.questions;
      cache.set(fileName, questions);
      return questions;
    },
  };
}

#### スケーリング処理

```javascript
/**
 * Canvasのスケーリング
 *
 * 処理:
 * 1. 親要素のサイズを取得
 * 2. アスペクト比 4:3 を維持しつつ最大サイズを計算
 * 3. デバイスピクセル比に対応（Retina等）
 *    - canvas.width = logicalWidth * devicePixelRatio
 *    - canvas.height = logicalHeight * devicePixelRatio
 *    - CSS width/height = 計算された表示サイズ
 *    - ctx.scale(devicePixelRatio, devicePixelRatio)
 * 4. window の resize イベントで再計算
 */
function resize(canvas, ctx) {
  const parent = canvas.parentElement;
  const parentWidth = parent.clientWidth;
  const parentHeight = parent.clientHeight;

  const scaleX = parentWidth / LOGICAL_WIDTH;
  const scaleY = parentHeight / LOGICAL_HEIGHT;
  const scale = Math.min(scaleX, scaleY);

  const displayWidth = Math.floor(LOGICAL_WIDTH * scale);
  const displayHeight = Math.floor(LOGICAL_HEIGHT * scale);
  const dpr = window.devicePixelRatio || 1;

  canvas.width = LOGICAL_WIDTH * dpr;
  canvas.height = LOGICAL_HEIGHT * dpr;
  canvas.style.width = `${displayWidth}px`;
  canvas.style.height = `${displayHeight}px`;

  // リサイズ時はscaleの累積を防ぐため、先にトランスフォームをリセット
  ctx.setTransform(1, 0, 0, 1, 0, 0);
  ctx.scale(dpr, dpr);
}

### 7.3 ゲームループ (game-loop.js)

```javascript
// presentation/canvas/game-loop.js

/**
 * requestAnimationFrame ベースのゲームループ
 *
 * 公開API:
 *   createGameLoop(updateFn, renderFn)
 *     .start()    - ループ開始
 *     .stop()     - ループ停止
 *     .pause()    - 一時停止（内部タイマー停止）
 *     .resume()   - 再開
 */

/**
 * @param {function(number): void} updateFn - 状態更新関数（deltaTime引数）
 * @param {function(): void} renderFn       - 描画関数
 */
export function createGameLoop(updateFn, renderFn) {
  let animationId = null;
  let lastTimestamp = 0;
  let running = false;

  function tick(timestamp) {
    if (!running) return;

    const deltaTime = lastTimestamp ? timestamp - lastTimestamp : 0;
    lastTimestamp = timestamp;

    updateFn(deltaTime);
    renderFn();

    animationId = requestAnimationFrame(tick);
  }

  return {
    start() {
      if (running) return;
      running = true;
      lastTimestamp = 0;
      animationId = requestAnimationFrame(tick);
    },

    stop() {
      running = false;
      if (animationId) {
        cancelAnimationFrame(animationId);
        animationId = null;
      }
    },

    pause() {
      this.stop();
    },

    resume() {
      this.start();
    },
  };
}

### 7.6 キーボードハンドラー (keyboard-handler.js)

```javascript
// presentation/input/keyboard-handler.js

/**
 * キーボード入力処理
 *
 * 公開API:
 *   createKeyboardHandler()
 *     .onInput(callback)    - 英字入力時のコールバック登録
 *     .onEscape(callback)   - Escキー押下時のコールバック登録
 *     .enable()             - 入力受付開始
 *     .disable()            - 入力受付停止
 *     .destroy()            - イベントリスナー解除
 */

/**
 * @returns {Object} KeyboardHandler
 */
export function createKeyboardHandler() {
  let inputCallback = null;
  let escapeCallback = null;
  let enabled = false;

  function handleKeyDown(event) {
    if (!enabled) return;

    // IME入力中は無視
    if (event.isComposing) return;

    // デフォルト動作を抑制（スペース等でスクロールさせない）
    if (event.key.length === 1 || event.key === 'Escape') {
      event.preventDefault();
    }

    if (event.key === 'Escape') {
      escapeCallback?.();
      return;
    }

    // 英字キーのみ受け付け
    const key = event.key.toLowerCase();
    if (/^[a-z]$/.test(key)) {
      inputCallback?.(key);
      return;
    }

    // ハイフン（長音）
    if (event.key === '-') {
      inputCallback?.('-');
      return;
    }

    // その他のキーは無視
  }

  document.addEventListener('keydown', handleKeyDown);

  return {
    onInput(callback) { inputCallback = callback; },
    onEscape(callback) { escapeCallback = callback; },
    enable() { enabled = true; },
    disable() { enabled = false; },
    destroy() {
      document.removeEventListener('keydown', handleKeyDown);
      inputCallback = null;
      escapeCallback = null;
    },
  };
}
```

```javascript
/**
 * Canvas上のボタン領域
 *
 * Canvasはネイティブのボタン要素を持たないため、
 * クリック座標からボタン領域を判定する。
 *
 * ボタン定義:
 *   { label, x, y, width, height, onClick }
 *
 * クリック処理:
 * 1. canvas の click イベントでクリック座標を取得
 * 2. スケーリングを逆算して論理座標に変換
 *    ```
 *    const rect = canvas.getBoundingClientRect();
 *    const logicalX = (event.clientX - rect.left) * (LOGICAL_WIDTH / rect.width);
 *    const logicalY = (event.clientY - rect.top) * (LOGICAL_HEIGHT / rect.height);
 *    ```
 * 3. 各ボタン領域と衝突判定
 *    ```
 *    function isInside(logicalX, logicalY, btn) {
 *      return logicalX >= btn.x && logicalX <= btn.x + btn.width
 *          && logicalY >= btn.y && logicalY <= btn.y + btn.height;
 *    }
 *    ```
 * 4. 一致したボタンの onClick を呼ぶ
 *
 * ホバー処理:
 * 1. mousemove でカーソル位置を追跡
 * 2. ボタン領域内なら cursor = 'pointer'
 */
```

  });
});

describe('入力記録', () => {
  test('正解でcomboが増加', () => {
    const session = createGameSession(mockConfig, mockQuestions);
    startSession(session);
    recordCorrect(session);
    recordCorrect(session);
    expect(session.combo).toBe(2);
    expect(session.correctCount).toBe(2);
  });

  test('ミスでcomboがリセット', () => {
    const session = createGameSession(mockConfig, mockQuestions);
    startSession(session);
    recordCorrect(session);
    recordCorrect(session);
    recordMiss(session);

#### infrastructure テスト補足

`JsonQuestionRepository` のテストは `globalThis.fetch` をモックして実施する:

```javascript
// infrastructure/repositories/__tests__/json-question-repository.test.js

import { vi, describe, test, expect, beforeEach } from 'vitest';
import { createJsonQuestionRepository } from '../json-question-repository.js';

describe('JsonQuestionRepository', () => {
  beforeEach(() => {
    vi.restoreAllMocks();
  });

  test('正常取得でデータを返す', async () => {
    const mockData = { questions: [{ id: 'test-001', display: 'テスト', reading: 'てすと' }] };
    globalThis.fetch = vi.fn().mockResolvedValue({
      ok: true,
      json: () => Promise.resolve(mockData),
    });
    const repo = createJsonQuestionRepository();
    const result = await repo.fetchQuestions('ja-romaji', 'easy');
    expect(result).toEqual(mockData.questions);
  });

  test('2回目はキャッシュを返す', async () => {
    const mockData = { questions: [{ id: 'test-001', display: 'テスト', reading: 'てすと' }] };
    globalThis.fetch = vi.fn().mockResolvedValue({
      ok: true,
      json: () => Promise.resolve(mockData),
    });
    const repo = createJsonQuestionRepository();
    await repo.fetchQuestions('ja-romaji', 'easy');
    await repo.fetchQuestions('ja-romaji', 'easy');
    expect(globalThis.fetch).toHaveBeenCalledTimes(1);
  });

  test('fetch失敗時にエラーを投げる', async () => {
    globalThis.fetch = vi.fn().mockResolvedValue({ ok: false, status: 404 });
    const repo = createJsonQuestionRepository();
    await expect(repo.fetchQuestions('ja-romaji', 'easy')).rejects.toThrow();
  });
});
